Table Image_Object {
  image_id integer [primary key]
  user_interactions char
}

Table Image_Tags {
  tag_id integer [primary key]
  metadata char
}

Table Image_Metadata {
  image_id integer [ref: - Image_Object.image_id]
  image_size char
  blob_storage_uuid char
  image_source_url char
  upload_date char
  status char
}

Table Image_Tag_Junction {
  image_id integer [ref: - Image_Object.image_id]
  tag_id integer [ref: - Image_Tags.tag_id]
  Indexes {
    (image_id, tag_id)
  }
}
